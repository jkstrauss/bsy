﻿ברכו וכו׳ בא״י אמ״ה אשר בדברו מעריב ערבים וכו׳ עד תמיד ימלך עלינו לעולם ועד
לֵיל שִׁמּוּרִים לְאִמֵּץ חֲלוּשֵׁי יָדַֽיִם
בִּטֵּא וְהוֹצֵאתִי אֶתְכֶם מִתַּֽחַת סִבְלוֹת מִצְרָֽיִם
גָּזַר מוֹעֲדִים וּזְמַנִּים מִיגִיעַת כַּפָּֽיִם
דָּת־הַיּוֹם לְקַדֵּשׁ עֲלֵי יַֽיִן בָּעַרְבָּֽיִם
בא״י המעריב ערבים

אהבת עולם וכו׳
לֵיל שִׁמּוּרִים לְהִתִּיר כִּבְלֵי תָם
וִעֵד וְהִצַּלְתִּי אֶתְכֶם מֵעֲבֹדָתָם
זְכֹר יְמוֹת עוֹלָם בִּֽינוּ דוֹרֹתָם
חָפְשִׁים עִבְרִים לְעוֹלָם וּלְעוֹלָם אַהֲבָתָם
בא״י אוהב עמו ישראל

שמע וגו׳ והיה וגו׳ ויאמר וגו׳ אמת ואמונה וכו׳ עד לך ענו שירה
פֶּֽסַח אוֺתוֺתָיו נִצְטַוּוּ לִשְׁמוֺר׃ יִשְׂרָאֵל עוֺדוֺ עֶֽבֶד חֲמוֺר׃ בְּאֶֽרֶץ מִצְרַֽיִם לֵאמֹר׃
פֶּֽסַח בְּיוֺם חִדּוּשׁ לְבָנָה׃ ווּעַד עֲשִׂיָּתוֺ לַאֲנִי יְשֵׁנָה׃ רִאשׁוֺן הוּא לָכֶם לְחָדְשֵׁי הַשָּׁנָה׃
פֶּֽסַח גְּבֶֽרֶת רַעֲנַנָה כַזַּֽיִת׃ סָבְלָה בֶּעָשׂוֺר עֵינֶֽיהָ שַֽׁיִת׃ שֶׂה לְבֵית אָבוֺת שֶׂה לַבָּֽית׃
פֶּֽסַח דִּבְחָתֵיהּ טֶֽרֶם יֵעָשֶׂה׃ פּוֹטְרִים וְנִמְשָׁכִים מִהְיוֹת מִשֶּׂה׃ לְפִי אָכְלוֹ תָּכֹֽסּוּ עַל הַשֶּׂה׃
פֶּֽסַח הִתְהַלֵּךְ שֶׂה תָמִים בָּאָֽחוּ׃ קָרְבְּנוֺתֵיכֶם כָּכֶם נוֹֽעַר יְשֻׁבָּֽחוּ׃ מִן הַכְּבָשִׂים וּמִן הָעִזִּים תִּקָּֽחוּ׃
פֶּֽסַח וְהָיָה בִּזְמַן עֵרוּב יוֹמַֽיִם׃ לֵב נֶֽפֶשׁ וּמְאֹד תִּמְסְרוּ לַשָּׁמַֽיִם׃ קְהַל עֲדַת יִשְׂרָאֵל בֵּין הָעַרְבָּֽיִם׃
פֶּֽסַח זְבִיחָה הֶכְשֵׁר סְעוּדַתְכֶם׃ וּשְׁמַרְתֶּם הַמִּצְוֹות וָחַי בָּהֶם׃ אֲשֶׁר יֹאכְלוּ אוֹתוֹ בָהֶם׃
פֶּֽסַח חֲזוֹן בְּתָרִים בִּשְּׂרוּֽהוּ׃ נָכְרִיּוּת וְעַבְדּוּת יְקְרֵֽהוּ׃ עַל מְרוֹרִים יֹאכְלֻֽהוּ׃
פֶּֽסַח טֶֽרֶם גָּמְרוֹ אַל טַלְטֵל בּוֹ׃ יִבָּדֵל דָּבָר אֲחֵר מְמְּסִבּוֹ׃ רֹאשׁוֺ עַל כְּרָעָיו וְעַל קִרְבּוֹ׃
פֶּֽסַח יֵאָכֵל שֶׁשְּׁחִיטָה תִּצְטָרְפוּ׃ מָחָר בַּל תִּשְׂרְפֶֽנּוּ אַף שֶׁתֹּאפוּ׃ עַד בֹּֽקֶר בָּאֵשׁ תִּשְׂרֹֽפוּ׃
פֶּֽסַח כָּֽכָה תֹּאכֲלֶֽנּוּ עֵדָי׃ וְתֶחֶפְזוּ וְאַל תֶּחֶטְאוּ לֵאמֹר דַּי׃ פֶּֽסַח הוּא לַיְֹיָ׃
פֶּֽסַח לֹא מַלְאָךְ עוֺבֵר בִּלְעָדַי׃ סוּס וְרוֺכְבוֺ יִגְלוּ סוֺדָי׃ אֲנִי יְֹיָ׃

לֵיל שִׁמּוּרִים לְטִאטֵא נֹף וּמִפְלָצוֹת
יָרָה בָּעֶֽרֶב תֹּאכְלוּ מַצּוֹת
כַּלְכָּלָתָם עֲנִיָּה נַפְשָׁם בִּלְתִּי קְרוּצוֹת
לֹא עֲבְדֵי פַרְעֹה רוּחוֹת דוֹצוֹת
בגילה ברנה בשמחה רבה ואמרו כלם

מי כמוכה וגו׳ מלכותך וכו׳ עד לפני משה
לֵיל שִׁמּוּרִים לְמֵישָׁרִים מַעֲשֵֽׂהוּ
נָם עַל מְרוֹרִים יֹאכְלוּֽהוּ
סָגוֺף יִסָּגֵף עוֺד לְדָרְשֵֽׁהוּ
עַנֵּה יְעַנֶּה אוֺתָֽנוּ וְנַמְלִיכֵֽהוּ
זה צור ישענו פצו פה ואמרו

יי מלך וגו׳ ונאמר וכו׳ עד מיד חזק ממנו
לֵיל שִׁמּוּרִים לְפֹקֵד גֶּֽפֶן קְלוּיָה
צָוַח וְגָאַלְתִּי אֶתְכֶם בִּזְרוֺֽעַ נְטוּיָה
קָרוֹב לְקוֹרְאָיו בְּאֶֽרֶץ מַאְפֵּלְיָה
רִיבָה רִיבִי וּגְאָלֵֽנִי עָנָֽנִי בַּמֶרְחַבְיָה
בא״י גאל ישׁראל

השכיבנו וכו׳ עד ופרש עלינו סכת שלומיך
לֵיל שִׁמּוּרִים לְשׁוֺמֵר בְּרִית אָבוֺת לְזַרְעָם
תִּנֵּא וְלָקַחְתִּי אֶתְכֶם לִי לְעָם
חֲסִידָיו זֵֽכֶר קָדְשׁוֺ יְזַמְּרוּ וְיִנְעָם
וְאֶל מְרוֺם צִיּוֺן בְּשָלוֺם יְבִיאֵם מוֺשִׁיעָם
בא״י פורש סכת שלום עלינו סכת שלום עלינו ועל כל עמו ישראל ועל ירושלם
